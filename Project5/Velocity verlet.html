void VV(int N,double final_time,double &x_initial,double &v_initial, double mass1, double mass2, ofstream &file){

    // Velocity-Verlet


    double h = final_time/((double) N);

    double x[N];

    double v[N];

    double t;


    x[0] = x_initial;       // initial position

    v[0] = v_initial;       // initial velocity

    x[1] = x[0] + v[0]*h*h;


    for(int i=1; i<N; i+=1){

        t = i*h;

        x[i+1] = 2*x[i] - x[i-1] + dvdt(x[i],v[i],mass1,mass2)*h*h;

        v[i] = (x[i+1] - x[i-1])/(2*h);


        file << t << "\t" << x[i] << "\t" << v[i] << endl; // save values in file

    }

}
